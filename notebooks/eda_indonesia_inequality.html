chart_theme <- function() {
  theme_minimal(base_size = 11) +
    theme(
      plot.title = element_text(face = "bold", size = 14, color = palette[["dark"]]),
      plot.subtitle = element_text(size = 10, color = palette[["gray"]]),
      axis.title = element_text(size = 10, color = palette[["dark"]]),
      axis.text = element_text(size = 9, color = palette[["dark"]]),
      legend.title = element_text(size = 9, face = "bold"),
      legend.position = "bottom",
      panel.grid.minor = element_blank()
    )
}

read_processed_csv <- function(filename) {
  read_csv(file.path(data_dir, filename), show_col_types = FALSE) %>%
    rename_with(str_trim)
}

Main Story

The storyline is: Indonesia is perceived as increasingly unequal, real GDP per capita has risen, average consumption improved across broad groups, but the distribution of gains shifted toward higher-income groups. The final section asks whether households feel better off once wages are compared with prices.

Nominal Wages, Prices, And Purchasing Power

wage_price_start_year <- 2000

wage_price_index <- data$labor %>%
  select(year, avg_wage) %>%
  left_join(data$prices %>% select(year, cpi_total), by = "year") %>%
  filter(year >= wage_price_start_year) %>%
  drop_na(avg_wage, cpi_total) %>%
  mutate(real_wage = avg_wage / cpi_total * 100) %>%
  mutate(
    nominal_wage_index = avg_wage / first(avg_wage) * 100,
    cpi_index = cpi_total / first(cpi_total) * 100,
    real_wage_index = real_wage / first(real_wage) * 100
  ) %>%
  select(year, nominal_wage_index, cpi_index, real_wage_index) %>%
  pivot_longer(
    cols = ends_with("_index"),
    names_to = "metric",
    values_to = "index"
  ) %>%
  mutate(metric = recode(
    metric,
    nominal_wage_index = "Nominal wage",
    cpi_index = "Consumer prices",
    real_wage_index = "Real wage"
  ))

wage_price_index_labels <- wage_price_index %>%
  group_by(metric) %>%
  arrange(year, .by_group = TRUE) %>%
  slice_tail(n = 1) %>%
  ungroup() %>%
  mutate(
    label_x = year + 0.28,
    label_y = case_when(
      metric == "Nominal wage" ~ index + 8,
      metric == "Consumer prices" ~ index - 2,
      metric == "Real wage" ~ index - 2,
      TRUE ~ index
    ),
    label = str_glue("{metric}  +{percent(index / 100 - 1, accuracy = 1)}")

Household Price Pressure

household_analysis_start_year <- 2020
household_analysis_end_year <- 2023

category_cpi <- data$prices %>%
  select(year, cpi_food, cpi_transport, cpi_housing, cpi_education, cpi_healthcare, cpi_restaurant) %>%
  pivot_longer(
    cols = starts_with("cpi_"),
    names_to = "category",
    values_to = "index"
  ) %>%
  drop_na(index) %>%
  filter(year >= household_analysis_start_year) %>%
  mutate(category = recode(
    category,
    cpi_food = "Food, beverages, tobacco",
    cpi_transport = "Transport",
    cpi_housing = "Housing, water, electricity, fuel",
    cpi_education = "Education",
    cpi_healthcare = "Health",
    cpi_restaurant = "Restaurants"
  ))

middle_consumption_pressure <- data$welfare_distribution %>%
  filter(
    group == "Middle 40%",
    between(year, household_analysis_start_year, household_analysis_end_year)
  ) %>%
  select(year, avg_welfare_ppp_daily) %>%
  left_join(data$prices %>% select(year, cpi_total), by = "year") %>%
  arrange(year) %>%
  drop_na(avg_welfare_ppp_daily, cpi_total) %>%
  mutate(
    real_consumption_index = avg_welfare_ppp_daily / first(avg_welfare_ppp_daily) * 100,
    cpi_index = cpi_total / first(cpi_total) * 100,
    metric = "Middle 40% average consumption",
    value = real_consumption_index * cpi_index / 100
  ) %>%
  select(year, metric, value)

price_pressure <- bind_rows(
  middle_consumption_pressure,
  category_cpi %>%
    transmute(year, metric = category, value = index)
) %>%
  filter(between(year, household_analysis_start_year, household_analysis_end_year)) %>%
  drop_na(value) %>%
  mutate(metric = factor(
    metric,
    levels = c(
      "Middle 40% average consumption",
      "Food, beverages, tobacco",
      "Transport",
      "Housing, water, electricity, fuel",
      "Education",
      "Health",
      "Restaurants"
    )
  )) %>%
  group_by(metric) %>%
  arrange(year, .by_group = TRUE) %>%
  mutate(
    base_value = first(value),
    index_start = value / base_value * 100,
    emphasis = if_else(metric == "Middle 40% average consumption", "highlight", "muted"),
    line_type = if_else(emphasis == "highlight", "solid", "dashed")
  ) %>%
  ungroup()

price_pressure_growth <- price_pressure %>%
  group_by(metric) %>%
  arrange(year, .by_group = TRUE) %>%
  summarise(
    growth_pct = last(value) / first(value) - 1,
    .groups = "drop"
  )

price_pressure_labels <- price_pressure %>%
  group_by(metric) %>%
  slice_max(year, n = 1, with_ties = FALSE) %>%
  ungroup() %>%
  left_join(price_pressure_growth, by = "metric") %>%
  mutate(
    label_x = household_analysis_end_year + 0.06,
    label_y = case_when(
      metric == "Transport" ~ 118.6,
      metric == "Food, beverages, tobacco" ~ 115.8,
      metric == "Restaurants" ~ 109.2,
      metric == "Middle 40% average consumption" ~ 108.1,
      metric == "Health" ~ 106.6,
      metric == "Education" ~ 105.7,
      metric == "Housing, water, electricity, fuel" ~ 104.8,
      TRUE ~ index_start
    ),
    label = str_glue("{metric}  {percent(growth_pct, accuracy = 0.1)}")
  )

price_pressure %>%
