chart_theme <- function() {
  theme_minimal(base_size = 11) +
    theme(
      plot.title = element_text(face = "bold", size = 14, color = palette[["dark"]]),
      plot.subtitle = element_text(size = 10, color = palette[["gray"]]),
      axis.title = element_text(size = 10, color = palette[["dark"]]),
      axis.text = element_text(size = 9, color = palette[["dark"]]),
      legend.title = element_text(size = 9, face = "bold"),
      legend.position = "bottom",
      panel.grid.minor = element_blank()
    )
}

read_processed_csv <- function(filename) {
  read_csv(file.path(data_dir, filename), show_col_types = FALSE) %>%
    rename_with(str_trim)
}

coverage <- indonesia_data %>%
  imap_dfr(function(df, dataset_name) {
    value_cols <- df %>%
      select(-any_of(c("year", "source"))) %>%
      names()

    df %>%
      rowwise() %>%
      mutate(
        dataset = dataset_name,
        available_fields = sum(!is.na(pick(all_of(value_cols)))),
        total_fields = length(value_cols),
        coverage_rate = if_else(total_fields > 0, available_fields / total_fields, 0)
      ) %>%
      ungroup() %>%
      select(dataset, year, available_fields, total_fields, coverage_rate)
  })

coverage %>%
  ggplot(aes(year, dataset, fill = coverage_rate)) +
  geom_tile(color = "white", linewidth = 0.25) +

availability <- indonesia_data %>%
  imap_dfr(function(df, dataset_name) {
    df %>%
      select(-any_of(c("year", "source"))) %>%
      summarise(across(everything(), ~ sum(!is.na(.x)))) %>%
      pivot_longer(everything(), names_to = "field", values_to = "non_missing_count") %>%
      mutate(
        dataset = dataset_name,
        total_rows = nrow(df),
        status = case_when(
          non_missing_count == 0 ~ "Empty",
          non_missing_count == total_rows ~ "Complete",
          TRUE ~ "Partly missing"
        )
      )
  }) %>%
  mutate(status = factor(status, levels = c("Complete", "Partly missing", "Empty")))

  arrange(dataset, status, field)

decile_levels <- c(
  "0 - 10%", "11 - 20%", "21 - 30%", "31 - 40%", "41 - 50%",
  "51 - 60%", "61 - 70%", "71 - 80%", "81 - 90%", "91 - 100%"
)

pip_percentiles <- read_csv(
  file.path(project_root, "data", "raw", "worldbank_pip_percentiles_indonesia.csv"),
  show_col_types = FALSE
) %>%
  filter(
    country_code == "IDN",
    reporting_level == "national"
  )

latest_available_consumption_year <- max(pip_percentiles$year, na.rm = TRUE)
long_horizon_start_year <- 2000
long_horizon_end_year <- min(latest_available_consumption_year, 2024)

ppp_base_year <- 2017

ppp_2017_conversion <- data$gdp %>%
  filter(year == ppp_base_year) %>%
  pull(ppp_private_consumption)

cpi_2017 <- data$prices %>%
  filter(year == ppp_base_year) %>%
  pull(cpi_total)

long_horizon_cpi <- data$prices %>%
  filter(year <= long_horizon_end_year, !is.na(cpi_total)) %>%
  slice_max(year, n = 1, with_ties = FALSE)

long_horizon_cpi_year <- long_horizon_cpi$year
long_horizon_price_conversion <- ppp_2017_conversion * long_horizon_cpi$cpi_total / cpi_2017

decile_consumption_long <- pip_percentiles %>%
  filter(between(year, long_horizon_start_year, long_horizon_end_year)) %>%
  mutate(
    percentile = as.integer(percentile),
    decile = case_when(
      between(percentile, 1, 10) ~ "0 - 10%",
      between(percentile, 11, 20) ~ "11 - 20%",
      between(percentile, 21, 30) ~ "21 - 30%",
      between(percentile, 31, 40) ~ "31 - 40%",
      between(percentile, 41, 50) ~ "41 - 50%",
      between(percentile, 51, 60) ~ "51 - 60%",
      between(percentile, 61, 70) ~ "61 - 70%",
      between(percentile, 71, 80) ~ "71 - 80%",
      between(percentile, 81, 90) ~ "81 - 90%",
      between(percentile, 91, 100) ~ "91 - 100%"
    )
  ) %>%
  drop_na(decile, avg_welfare, pop_share) %>%
  group_by(year, decile) %>%
  summarise(
    avg_consumption = weighted.mean(avg_welfare, pop_share),
    .groups = "drop"
  ) %>%
  mutate(decile = factor(decile, levels = decile_levels)) %>%
  arrange(decile, year)

cpi_index <- data$prices %>%
  filter(between(year, long_horizon_start_year, long_horizon_end_year)) %>%
  arrange(year) %>%
  mutate(cpi_index_2000 = cpi_total / first(cpi_total) * 100) %>%
  select(year, cpi_index_2000)

decile_consumption_index <- decile_consumption_long %>%
  group_by(decile) %>%
  arrange(year, .by_group = TRUE) %>%
  mutate(real_consumption_index_2000 = avg_consumption / first(avg_consumption) * 100) %>%
  ungroup() %>%
  left_join(cpi_index, by = "year") %>%
  transmute(
    year,
    series = as.character(decile),
    index_2000 = real_consumption_index_2000 * cpi_index_2000 / 100
  )

index_line_palette <- set_names(hcl.colors(length(decile_levels), palette = "Dark 3"), decile_levels)
decile_consumption_long %>%
  group_by(decile) %>%
  arrange(year, .by_group = TRUE) %>%
  summarise(
    consumption_2000 = first(avg_consumption),
    latest_consumption = last(avg_consumption),
    growth = latest_consumption / consumption_2000 - 1,
    .groups = "drop"
  ) %>%
  mutate(
    latest_consumption_rupiah = latest_consumption * long_horizon_price_conversion,
    consumption_label = str_glue("Rp {number(latest_consumption_rupiah, accuracy = 100, big.mark = ',')} / day"),
    label_vjust = if_else(growth >= 0, -0.35, 1.25)
  ) %>%
  ggplot(aes(decile, growth)) +
  geom_hline(yintercept = 0, color = palette[["gray"]], linetype = "dashed") +
  geom_col(fill = palette[["green"]], width = 0.72) +
  geom_text(
    aes(label = consumption_label, vjust = label_vjust),
    size = 3
  ) +
  scale_y_continuous(
    labels = percent_format(accuracy = 1),
    expand = expansion(mult = c(0.08, 0.14))
  ) +
  labs(
    title = "Long-Run Average Consumption Growth by Decile",
    subtitle = str_glue("Bar height shows {long_horizon_start_year}-{long_horizon_end_year} growth in PIP's constant 2017 PPP consumption welfare."),
    x = "Population decile", y = "Consumption growth",
    caption = str_glue("Labels show {long_horizon_end_year} consumption translated to rupiah/day in {long_horizon_cpi_year} prices: 2017 PPP welfare x 2017 private-consumption PPP x CPI growth since 2017.")
  ) +
  chart_theme() +
  theme(
    axis.text.x = element_text(angle = 35, hjust = 1),
    legend.position = "none"
  )
